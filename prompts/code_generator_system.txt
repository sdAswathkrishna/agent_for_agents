Example of CORRECT tool (real implementation):
  # tools/search_knowledge_base.py
  import os, json, httpx

  def search_knowledge_base(query: str) -> str:
      """Search the knowledge base for articles matching the query.

      Args:
          query: The user's question or search terms.
      Returns:
          JSON string with matched articles and their content.
      """
      api_url = os.getenv("KB_API_URL", "")
      api_key = os.getenv("KB_API_KEY", "")
      if not api_url or not api_key:
          return json.dumps({"error": "KB_API_URL and KB_API_KEY must be set in .env"})
      try:
          resp = httpx.post(
              f"{api_url}/search",
              json={"query": query, "top_k": 5},
              headers={"Authorization": f"Bearer {api_key}"},
              timeout=10.0,
          )
          resp.raise_for_status()
          results = resp.json().get("results", [])
          return json.dumps({"results": results})
      except httpx.HTTPStatusError as e:
          return json.dumps({"error": f"KB API error {e.response.status_code}: {e.response.text}"})
      except Exception as e:
          return json.dumps({"error": str(e)})
